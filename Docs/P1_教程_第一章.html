🎯 第一章目标

用纯C++搭建一个UE5第三人称角色，能移动、看、跳跃、攻击、冲刺，并且有对应的动画表现。

搭建顺序：Build.cs → 骨架 → 输入 → 动画 → 冲刺，每一步都编译验证。

Step 0：修改 Build.cs ⚠️必须第一步做！

⚠️ 踩坑经验：如果不先加 PublicIncludePaths，后面用子文件夹组织代码时编译会报 fatal error C1083: No such file or directory

0.1 加 include 路径

PublicIncludePaths.AddRange(new string[] { ModuleDirectory });

告诉编译器："我的子文件夹里的头文件也要能被找到"。

0.2 加模块依赖

PublicDependencyModuleNames.AddRange(new string[] { 
    "Core", "CoreUObject", "Engine", "InputCore", "EnhancedInput" 
});

EnhancedInput 是UE5新输入系统的模块，不加后面用增强输入会编译报错。

完整文件 ue5_battle.Build.cs

using UnrealBuildTool;

public class ue5_battle : ModuleRules
{
    public ue5_battle(ReadOnlyTargetRules Target) : base(Target)
    {
        PCHUsage = PCHUsageMode.UseExplicitOrSharedPCHs;
        PublicIncludePaths.AddRange(new string[] { ModuleDirectory });
        PublicDependencyModuleNames.AddRange(new string[] { 
            "Core", "CoreUObject", "Engine", "InputCore", "EnhancedInput" 
        });
    }
}

1. Ctrl+Alt+F11（最推荐）
2. 顶部菜单 工具→编译
3. 关闭编辑器重开（最稳妥）

Step 3：BattleCharacter 骨架 — 角色本体

BattleCharacter.h（骨架版）

#pragma once
#include "CoreMinimal.h"
#include "GameFramework/Character.h"
#include "BattleCharacter.generated.h"

class USpringArmComponent;
class UCameraComponent;

UCLASS()
class UE5_BATTLE_API ABattleCharacter : public ACharacter
{
    GENERATED_BODY()
public:
    ABattleCharacter();
    virtual void BeginPlay() override;
    virtual void SetupPlayerInputComponent(class UInputComponent* PlayerInputComponent) override;

protected:
    UPROPERTY(VisibleAnywhere, BlueprintReadOnly, Category = Camera)
    USpringArmComponent* CameraBoom;

    UPROPERTY(VisibleAnywhere, BlueprintReadOnly, Category = Camera)
    UCameraComponent* FollowCamera;
};

BattleCharacter.cpp 构造函数

ABattleCharacter::ABattleCharacter()
{
    // 1. 加载骨骼网格体
    static ConstructorHelpers::FObjectFinder<USkeletalMesh> MeshAsset(
        TEXT("/Game/MyResources/Meshes/SKM_Manny"));
    if (MeshAsset.Succeeded())
    {
        GetMesh()->SetSkeletalMesh(MeshAsset.Object);
        GetMesh()->SetRelativeLocation(FVector(0.f, 0.f, -90.f));
        GetMesh()->SetRelativeRotation(FRotator(0.f, -90.f, 0.f));
    }

    // 2. 创建弹簧臂
    CameraBoom = CreateDefaultSubobject<USpringArmComponent>(TEXT("CameraBoom"));
    CameraBoom->SetupAttachment(RootComponent);
    CameraBoom->TargetArmLength = 300.0f;
    CameraBoom->bUsePawnControlRotation = true;

    // 3. 创建跟随摄像机
    FollowCamera = CreateDefaultSubobject<UCameraComponent>(TEXT("FollowCamera"));
    FollowCamera->SetupAttachment(CameraBoom, USpringArmComponent::SocketName);

    // 4. 角色旋转设置（朝向由移动方向决定）
    bUseControllerRotationPitch = false;
    bUseControllerRotationYaw = false;
    bUseControllerRotationRoll = false;

    // 5. 移动组件参数
    GetCharacterMovement()->bOrientRotationToMovement = true;
    GetCharacterMovement()->RotationRate = FRotator(0.f, 500.f, 0.f);
    GetCharacterMovement()->MaxWalkSpeed = 500.0f;
    GetCharacterMovement()->JumpZVelocity = 700.0f;
    GetCharacterMovement()->AirControl = 0.35f;
}
⚠️ .h里声明的变量名叫 CameraBoom，.cpp里就必须用一模一样的名字，不能写成 SpringArm，否则编译报错 error C2065。

⚠️ 关键配置：IA_Sprint 和 IA_Jump 的 Triggers（触发器）数组必须清空，不要加任何触发器！默认行为就是"按住持续触发Triggered，松开触发Completed"。如果加了Down触发器，冲刺会只触发一次就停。

Step 5：输入绑定 — 让角色动起来

⚠️ 踩坑经验（加载顺序）：构造函数里必须先加载 DA_BattleInputConfig（得到InputConfig），再加载 IMC_Default 赋值。如果顺序反了，InputConfig还是nullptr，赋值会失败！

构造函数加载资产（正确顺序）

// 先加载 InputConfig（必须先！）
static ConstructorHelpers::FObjectFinder<UBattleInputConfig> InputConfigAsset(
    TEXT("/Game/MyResources/Input/DA_BattleInputConfig"));
if (InputConfigAsset.Succeeded())
    InputConfig = InputConfigAsset.Object;

// 再加载 IMC（InputConfig已有值才能赋值）
static ConstructorHelpers::FObjectFinder<UInputMappingContext> IMCAsset(
    TEXT("/Game/MyResources/Input/IMC_Default"));
if (InputConfig && IMCAsset.Succeeded())
    InputConfig->InputMappingContext = IMCAsset.Object;

BeginPlay — 注册 IMC

void ABattleCharacter::BeginPlay()
{
    Super::BeginPlay();
    APlayerController* PC = Cast<APlayerController>(Controller);
    if (PC && InputConfig)
    {
        auto* Subsystem = ULocalPlayer::GetSubsystem
            <UEnhancedInputLocalPlayerSubsystem>(PC->GetLocalPlayer());
        if (Subsystem && InputConfig->InputMappingContext)
            Subsystem->AddMappingContext(InputConfig->InputMappingContext, 0);
    }
}

SetupPlayerInputComponent — 绑定输入

UEnhancedInputComponent* EIC = Cast<UEnhancedInputComponent>(PlayerInputComponent);

// 移动（持续触发）
EIC->BindAction(InputConfig->InputActions->MoveAction,
    ETriggerEvent::Triggered, this, &ABattleCharacter::Move);
// 视角（持续触发）
EIC->BindAction(InputConfig->InputActions->LookAction,
    ETriggerEvent::Triggered, this, &ABattleCharacter::Look);
// 跳跃（按下+松开）
EIC->BindAction(InputConfig->InputActions->JumpAction,
    ETriggerEvent::Started, this, &ABattleCharacter::StartJump);
EIC->BindAction(InputConfig->InputActions->JumpAction,
    ETriggerEvent::Completed, this, &ABattleCharacter::StopJump);

输入处理函数

void ABattleCharacter::Move(const FInputActionValue& Value)
{
    FVector2D MovementVector = Value.Get<FVector2D>();
    if (Controller && MovementVector.SizeSquared() > 0.0f)
    {
        const FRotator Rotation = Controller->GetControlRotation();
        const FRotator YawRotation(0, Rotation.Yaw, 0);
        const FVector Forward = FRotationMatrix(YawRotation).GetUnitAxis(EAxis::X);
        const FVector Right = FRotationMatrix(YawRotation).GetUnitAxis(EAxis::Y);
        AddMovementInput(Forward, MovementVector.Y);
        AddMovementInput(Right, MovementVector.X);
    }
}

void ABattleCharacter::Look(const FInputActionValue& Value)
{
    FVector2D LookVector = Value.Get<FVector2D>();
    if (Controller)
    {
        AddControllerYawInput(LookVector.X);
        AddControllerPitchInput(LookVector.Y);
    }
}

void ABattleCharacter::StartJump() { Jump(); }
void ABattleCharacter::StopJump()  { StopJumping(); }

Step 6：动画系统 — 让角色有动画表现

BattleAnimInstance.h

UCLASS()
class UE5_BATTLE_API UBattleAnimInstance : public UAnimInstance
{
    GENERATED_BODY()
public:
    virtual void NativeUpdateAnimation(float DeltaSeconds) override;
protected:
    UPROPERTY(BlueprintReadOnly) float Speed = 0.0f;
    UPROPERTY(BlueprintReadOnly) bool bIsInAir = false;
    UPROPERTY(BlueprintReadOnly) bool bIsMoving = false;
    UPROPERTY(BlueprintReadOnly) bool bIsAttacking = false;
    UPROPERTY(BlueprintReadOnly) bool bIsSprintingAnim = false;
};

BattleAnimInstance.cpp

void UBattleAnimInstance::NativeUpdateAnimation(float DeltaSeconds)
{
    Super::NativeUpdateAnimation(DeltaSeconds);
    ACharacter* Character = Cast<ACharacter>(GetOwningActor());
    if (!Character) return;
    
    auto* MoveComp = Character->GetCharacterMovement();
    if (!MoveComp) return;

    Speed = Character->GetVelocity().Size();
    bIsInAir = MoveComp->IsFalling();
    bIsMoving = Speed > 3.0f;

    ABattleCharacter* BattleChar = Cast<ABattleCharacter>(Character);
    if (BattleChar)
    {
        auto* Montage = BattleChar->GetAttackMontage();
        bIsAttacking = Montage && Montage_IsPlaying(Montage);
        if (BattleChar->BattleMovement)
            bIsSprintingAnim = BattleChar->BattleMovement->IsSprinting();
    }
}

构造函数加载动画蓝图

static ConstructorHelpers::FClassFinder<UAnimInstance> AnimBPAsset(
    TEXT("/Game/MyResources/Animation/ABP_Battle"));
if (AnimBPAsset.Succeeded())

EIC->BindAction(InputConfig->InputActions->AttackAction,
    ETriggerEvent::Started, this, &ABattleCharacter::Attack);

// 实现
void ABattleCharacter::Attack()
{
    if (AttackMontage)
    {
        UAnimInstance* AnimInstance = GetMesh()->GetAnimInstance();
        if (AnimInstance && !AnimInstance->Montage_IsPlaying(AttackMontage))
            AnimInstance->Montage_Play(AttackMontage, 1.0f);
    }
}

Step 8：自定义移动组件 + 冲刺 ⚠️重点难点

BattleCharacterMovementComponent.h

UCLASS()
class UE5_BATTLE_API UBattleCharacterMovementComponent 
    : public UCharacterMovementComponent
{
    GENERATED_BODY()
public:
    UBattleCharacterMovementComponent();
    void SetWantsToSprint(bool bSprint);
    bool IsSprinting() const { return bIsSprinting; }
    void StartSprinting();
    void StopSprinting();
    virtual void TickComponent(float DeltaTime, ELevelTick TickType,
        FActorComponentTickFunction* ThisTickFunction) override;
private:
    float WalkSpeed = 500.0f;
    float SprintSpeed = 900.0f;
    bool bWantsToSprint = false;
    bool bIsSprinting = false;
};

BattleCharacterMovementComponent.cpp

UBattleCharacterMovementComponent::UBattleCharacterMovementComponent()
{
    MaxWalkSpeed = WalkSpeed;
    MaxAcceleration = 2048.0f;
    GroundFriction = 8.0f;
    BrakingDecelerationWalking = 2048.0f;
}

void UBattleCharacterMovementComponent::TickComponent(...)
{
    Super::TickComponent(DeltaTime, TickType, ThisTickFunction);
    if (bWantsToSprint) {
        bIsSprinting = true;
        MaxWalkSpeed = SprintSpeed;
    } else {
        bIsSprinting = false;
        MaxWalkSpeed = WalkSpeed;
    }
}

void ...::StartSprinting() { SetWantsToSprint(true); }
void ...::StopSprinting()  { SetWantsToSprint(false); }

替换默认移动组件（⚠️关键写法）

⚠️ 踩坑经验：必须用 FObjectInitializer 在构造函数里替换！父类 ACharacter 在构造函数执行前就已经创建了默认移动组件，只有通过 ObjectInitializer 才能在创建前拦截并替换。参考 Lyra 的 ALyraCharacter 写法。
// .h 改构造函数签名
ABattleCharacter(const FObjectInitializer& ObjectInitializer);

// .cpp
ABattleCharacter::ABattleCharacter(const FObjectInitializer& ObjectInitializer)
    : Super(ObjectInitializer.SetDefaultSubobjectClass
        <UBattleCharacterMovementComponent>(
            ACharacter::CharacterMovementComponentName))
{
    // ... 原有代码 ...
    BattleMovement = Cast<UBattleCharacterMovementComponent>(
        GetCharacterMovement());
}

Step 9：已知问题 & 注意事项

🐛 已知 Bug

#Bug原因
1构造函数中IMC加载顺序错误IMC加载代码放在InputConfig赋值之前，if(InputConfig&&...)永远false
2构造函数修改共享DataAsset会污染原始资产，教学项目单实例不影响
3冲刺时跳跃不触发（已修复）混用IsInputKeyDown和增强输入导致按键被吞

⚠️ 编辑器必做配置清单

配置项说明
IA_Sprint 触发器清空不加任何Trigger，默认按住持续触发
IA_Jump 触发器清空同上
动画蓝图加 DefaultSlot否则蒙太奇播放无效果
IMC_Default 5个IA都映射好否则对应功能不生效
